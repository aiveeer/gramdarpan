    48→
    49→async function apiGet(table: string, search?: string) {
    50→  const params = new URLSearchParams({ table });
    51→  if (search) params.set('search', search);
    52→  const res = await fetch(`/api/master?${params}`);
    53→  if (!res.ok) throw new Error('Fetch failed');
    54→  return res.json();
    55→}
    56→
    57→async function apiCreate(table: string, data: Record<string, unknown>) {
    58→  const res = await fetch('/api/master', {
    59→    method: 'POST',
    60→    headers: { 'Content-Type': 'application/json' },
    61→    body: JSON.stringify({ table, data }),
    62→  });
    63→  const json = await res.json();
    64→  if (!res.ok) throw new Error(json.error || 'Create failed');
    65→  return json;
    66→}
    67→
    68→async function apiUpdate(table: string, data: Record<string, unknown>) {
    69→  const res = await fetch('/api/master', {
    70→    method: 'POST',
    71→    headers: { 'Content-Type': 'application/json' },
    72→    body: JSON.stringify({ table, data }),
    73→  });
    74→  const json = await res.json();
    75→  if (!res.ok) throw new Error(json.error || 'Update failed');
    76→  return json;
    77→}
    78→
    79→async function apiDelete(table: string, id: string) {
    80→  const res = await fetch(`/api/master?table=${table}&id=${id}`, { method: 'DELETE' });
    81→  if (!res.ok) throw new Error('Delete failed');
    82→  return res.json();
    83→}
    84→
    85→async function apiSeed(table: string) {
    86→  const res = await fetch('/api/master', {
    87→    method: 'POST',
    88→    headers: { 'Content-Type': 'application/json' },
    89→    body: JSON.stringify({ table, action: 'seed' }),
    90→  });
    91→  return res.json();
    92→}

   307→interface CrudListProps {
   308→  title: string;
   309→  table: string;
   310→  colorKey: TabColorKey;
   311→  icon: React.ReactNode;
   312→  columns: { key: string; label: string; render?: (item: Record<string, unknown>) => React.ReactNode }[];
   313→  formFields: FormField[];
   314→  seedOnEmpty?: boolean;
   315→  renderExtraForm?: (data: Record<string, unknown>, setData: (d: Record<string, unknown>) => void) => React.ReactNode;
   316→  renderExtraColumns?: (item: Record<string, unknown>) => React.ReactNode;
   317→  defaultFormData?: Record<string, unknown>;
   318→}
   319→
   320→interface FormField {
   321→  key: string;
   322→  label: string;
   323→  type?: 'text' | 'number' | 'select' | 'checkbox' | 'textarea';
   324→  required?: boolean;
   325→  placeholder?: string;
   326→  options?: { value: string; label: string }[];
   327→  colSpan?: number;
   328→}
   329→
   330→function CrudList({
   331→  title,
   332→  table,
   333→  colorKey,
   334→  icon,
   335→  columns,
   336→  formFields,
   337→  seedOnEmpty,
   338→  renderExtraForm,
   339→  renderExtraColumns,
   340→  defaultFormData,
   341→}: CrudListProps) {
   342→  const [items, setItems] = useState<Record<string, unknown>[]>([]);
   343→  const [loading, setLoading] = useState(true);
   344→  const [search, setSearch] = useState('');
   345→  const [dialogOpen, setDialogOpen] = useState(false);
   346→  const [editItem, setEditItem] = useState<Record<string, unknown> | null>(null);
   347→  const [formData, setFormData] = useState<Record<string, unknown>>(defaultFormData || {});
   348→  const [saving, setSaving] = useState(false);
   349→  const [deleting, setDeleting] = useState<string | null>(null);
   350→
   351→  const c = tabColors[colorKey];
   352→
   353→  const fetchData = useCallback(async () => {
   354→    setLoading(true);
   355→    try {
   356→      let result = await apiGet(table, search || undefined);
   357→      if (!Array.isArray(result)) result = result ? [result] : [];
   358→      if (result.length === 0 && seedOnEmpty) {
   359→        await apiSeed(table);
   360→        result = await apiGet(table, search || undefined);
   361→        if (!Array.isArray(result)) result = result ? [result] : [];
   362→      }
   363→      setItems(result);
   364→    } catch {
   365→      toast({ title: 'त्रुटी', description: 'डेटा लोड करण्यात अयशस्वी', variant: 'destructive' });
   366→    } finally {
   367→      setLoading(false);
   368→    }
   369→  }, [table, search, seedOnEmpty]);
   370→
   371→  useEffect(() => { fetchData(); }, [fetchData]);
   372→
   373→  const handleAdd = () => {
   374→    setEditItem(null);
   375→    setFormData(defaultFormData || {});
   376→    setDialogOpen(true);
   377→  };
   378→
   379→  const handleEdit = (item: Record<string, unknown>) => {
   380→    setEditItem(item);
   381→    const fd: Record<string, unknown> = {};
   382→    for (const f of formFields) {
   383→      fd[f.key] = item[f.key] ?? (f.type === 'checkbox' ? false : f.type === 'number' ? '' : '');
   384→    }
   385→    setFormData(fd);
   386→    setDialogOpen(true);
   387→  };
   388→
   389→  const handleDelete = async (id: string) => {
   390→    if (!confirm('हा रेकॉर्ड हटवायचा आहे का?')) return;
   391→    setDeleting(id);
   392→    try {
   393→      await apiDelete(table, id);
   394→      toast({ title: 'यशस्वी', description: 'रेकॉर्ड हटवला' });
   395→      fetchData();
   396→    } catch (err: unknown) {
   397→      toast({ title: 'त्रुटी', description: err instanceof Error ? err.message : 'हटवणे अयशस्वी', variant: 'destructive' });
   398→    } finally {
   399→      setDeleting(null);
   400→    }
   401→  };
   402→
   403→  const handleSave = async () => {
   404→    setSaving(true);
   405→    try {
   406→      const saveData: Record<string, unknown> = { ...formData };
   407→      if (editItem?.id) saveData.id = editItem.id;
   408→      for (const f of formFields) {
   409→        if (f.type === 'number' && saveData[f.key] !== undefined && saveData[f.key] !== '') {
   410→          saveData[f.key] = parseFloat(String(saveData[f.key])) || 0;
   411→        }
   412→        if (f.type === 'number' && (saveData[f.key] === '' || saveData[f.key] === undefined)) {
   413→          saveData[f.key] = null;
   414→        }
   415→      }
   416→      await (editItem?.id ? apiUpdate(table, saveData) : apiCreate(table, saveData));
   417→      toast({ title: 'यशस्वी', description: editItem?.id ? 'रेकॉर्ड अपडेट झाला' : 'नवीन रेकॉर्ड जोडला' });

   434→        icon={icon}

   437→        onSearchChange={setSearch}
