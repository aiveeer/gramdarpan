    54→
    55→async function apiGet(table: string, search?: string) {
    56→  const params = new URLSearchParams({ table });
    57→  if (search) params.set('search', search);
    58→  const res = await fetch(`/api/master?${params}`);
    59→  if (!res.ok) throw new Error('Fetch failed');
    60→  return res.json();
    61→}
    62→
    63→async function apiCreate(table: string, data: Record<string, unknown>) {
    64→  const res = await fetch('/api/master', {
    65→    method: 'POST',
    66→    headers: { 'Content-Type': 'application/json' },
    67→    body: JSON.stringify({ table, data }),
    68→  });
    69→  const json = await res.json();
    70→  if (!res.ok) throw new Error(json.error || 'Create failed');
    71→  return json;
    72→}
    73→
    74→async function apiUpdate(table: string, data: Record<string, unknown>) {
    75→  const res = await fetch('/api/master', {
    76→    method: 'POST',
    77→    headers: { 'Content-Type': 'application/json' },
    78→    body: JSON.stringify({ table, data }),
    79→  });
    80→  const json = await res.json();
    81→  if (!res.ok) throw new Error(json.error || 'Update failed');
    82→  return json;
    83→}
    84→
    85→async function apiDelete(table: string, id: string) {
    86→  const res = await fetch(`/api/master?table=${table}&id=${id}`, { method: 'DELETE' });
    87→  if (!res.ok) throw new Error('Delete failed');
    88→  return res.json();
    89→}
    90→
    91→async function apiSeed(table: string) {
    92→  const res = await fetch('/api/master', {
    93→    method: 'POST',
    94→    headers: { 'Content-Type': 'application/json' },
    95→    body: JSON.stringify({ table, action: 'seed' }),
    96→  });
    97→  return res.json();
    98→}

   329→  type?: 'text' | 'number' | 'select' | 'checkbox' | 'textarea';

   335→
   336→function CrudList({
   337→  title,
   338→  table,
   339→  colorKey,
   340→  icon,
   341→  columns,
   342→  formFields,
   343→  seedOnEmpty,
   344→  renderExtraForm,
   345→  renderExtraColumns,
   346→  defaultFormData,
   347→}: CrudListProps) {
   348→  const [items, setItems] = useState<Record<string, unknown>[]>([]);
   349→  const [loading, setLoading] = useState(true);
   350→  const [search, setSearch] = useState('');
   351→  const [dialogOpen, setDialogOpen] = useState(false);
   352→  const [editItem, setEditItem] = useState<Record<string, unknown> | null>(null);
   353→  const [formData, setFormData] = useState<Record<string, unknown>>(defaultFormData || {});
   354→  const [saving, setSaving] = useState(false);
   355→  const [deleting, setDeleting] = useState<string | null>(null);
   356→
   357→  const c = tabColors[colorKey];
   358→
   359→  const fetchData = useCallback(async () => {
   360→    setLoading(true);
   361→    try {
   362→      let result = await apiGet(table, search || undefined);
   363→      if (!Array.isArray(result)) result = result ? [result] : [];
   364→      if (result.length === 0 && seedOnEmpty) {
   365→        await apiSeed(table);
   366→        result = await apiGet(table, search || undefined);
   367→        if (!Array.isArray(result)) result = result ? [result] : [];
   368→      }
   369→      setItems(result);
   370→    } catch {
   371→      toast({ title: 'त्रुटी', description: 'डेटा लोड करण्यात अयशस्वी', variant: 'destructive' });
   372→    } finally {
   373→      setLoading(false);
   374→    }
   375→  }, [table, search, seedOnEmpty]);
   376→
   377→  useEffect(() => { fetchData(); }, [fetchData]);
   378→
   379→  const handleAdd = () => {
   380→    setEditItem(null);
   381→    setFormData(defaultFormData || {});
   382→    setDialogOpen(true);
   383→  };
   384→
   385→  const handleEdit = (item: Record<string, unknown>) => {
   386→    setEditItem(item);
   387→    const fd: Record<string, unknown> = {};
   388→    for (const f of formFields) {
   389→      fd[f.key] = item[f.key] ?? (f.type === 'checkbox' ? false : f.type === 'number' ? '' : '');
   390→    }
   391→    setFormData(fd);
   392→    setDialogOpen(true);
   393→  };
   394→
   395→  const handleDelete = async (id: string) => {
   396→    if (!confirm('हा रेकॉर्ड हटवायचा आहे का?')) return;
   397→    setDeleting(id);
   398→    try {
   399→      await apiDelete(table, id);
   400→      toast({ title: 'यशस्वी', description: 'रेकॉर्ड हटवला' });
   401→      fetchData();
   402→    } catch (err: unknown) {
   403→      toast({ title: 'त्रुटी', description: err instanceof Error ? err.message : 'हटवणे अयशस्वी', variant: 'destructive' });
   404→    } finally {
   405→      setDeleting(null);
   406→    }
   407→  };
   408→
   409→  const handleSave = async () => {
   410→    setSaving(true);
   411→    try {
   412→      const saveData: Record<string, unknown> = { ...formData };
   413→      if (editItem?.id) saveData.id = editItem.id;
   414→      for (const f of formFields) {
   415→        if (f.type === 'number' && saveData[f.key] !== undefined && saveData[f.key] !== '') {
   416→          saveData[f.key] = parseFloat(String(saveData[f.key])) || 0;
   417→        }
   418→        if (f.type === 'number' && (saveData[f.key] === '' || saveData[f.key] === undefined)) {
   419→          saveData[f.key] = null;
   420→        }
   421→      }
   422→      await (editItem?.id ? apiUpdate(table, saveData) : apiCreate(table, saveData));
   423→      toast({ title: 'यशस्वी', description: editItem?.id ? 'रेकॉर्ड अपडेट झाला' : 'नवीन रेकॉर्ड जोडला' });
   424→      setDialogOpen(false);
   425→      fetchData();
   426→    } catch (err: unknown) {
   427→      toast({ title: 'त्रुटी', description: err instanceof Error ? err.message : 'जतन अयशस्वी', variant: 'destructive' });
   428→    } finally {
   429→      setSaving(false);
   430→    }
   431→  };
   432→
   433→  const filteredItems = items;
   434→
   435→  return (
   436→    <Card className="overflow-hidden">
   437→      <SectionHeader
   438→        colorKey={colorKey}
   439→        title={title}
   440→        icon={icon}
   441→        count={filteredItems.length}
   442→        search={search}
   443→        onSearchChange={setSearch}
   444→        onRefresh={fetchData}
   445→        onAdd={handleAdd}
   446→        loading={loading}
   447→      />
   448→      <CardContent className={`${c.bg} p-4`}>
   449→        {loading ? (
   450→          <div className="text-center py-8 text-muted-foreground animate-pulse">लोड होत आहे...</div>
   451→        ) : filteredItems.length === 0 ? (
   452→          <div className="text-center py-8 text-muted-foreground">कोणतेही रेकॉर्ड नाहीत</div>
   453→        ) : (
   454→          <div className="max-h-96 overflow-y-auto border rounded-lg border-white/50 shadow-sm">
   455→            <Table>
   456→              <ColoredTableHeader colorKey={colorKey}>
   457→                <ColoredTableHead colorKey={colorKey} className="w-12">क्र.</ColoredTableHead>
   458→                {columns.map((col) => (
   459→                  <ColoredTableHead key={col.key} colorKey={colorKey}>{col.label}</ColoredTableHead>
   460→                ))}
   461→                <ColoredTableHead colorKey={colorKey} className="w-24 text-right">क्रिया</ColoredTableHead>
   462→              </ColoredTableHeader>
   463→              <TableBody>
   464→                {filteredItems.map((item, idx) => (
   465→                  <TableRow key={String(item.id)} className="hover:bg-white/60">
   466→                    <TableCell className="font-medium">{idx + 1}</TableCell>
   467→                    {columns.map((col) => (
   468→                      <TableCell key={col.key}>
   469→                        {col.render ? col.render(item) : String(item[col.key] ?? '-')}
   470→                      </TableCell>
   471→                    ))}
   472→                    {renderExtraColumns && <TableCell>{renderExtraColumns(item)}</TableCell>}
   473→                    <TableCell className="text-right">
   474→                      <div className="flex justify-end gap-1">
   475→                        <Button variant="ghost" size="sm" onClick={() => handleEdit(item)} className="hover:bg-white/60">
   476→                          <Pencil className="h-4 w-4" />
   477→                        </Button>
   478→                        <Button
   479→                          variant="ghost"
   480→                          size="sm"
   481→                          onClick={() => handleDelete(String(item.id))}
   482→                          disabled={deleting === String(item.id)}
   483→                          className="hover:bg-white/60"
   484→                        >
   485→                          {deleting === String(item.id) ? (
   486→                            <Loader2 className="h-4 w-4 animate-spin" />
   487→                          ) : (
   488→                            <Trash2 className="h-4 w-4 text-red-500" />
   489→                          )}
   490→                        </Button>
   491→                      </div>
   492→                    </TableCell>
   493→                  </TableRow>
   494→                ))}
   495→              </TableBody>
   496→            </Table>
   497→          </div>
   498→        )}
   499→      </CardContent>
   500→
   501→      {/* Add/Edit Dialog */}
   502→      <Dialog open={dialogOpen} onOpenChange={setDialogOpen}>
   503→        <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
   504→          <DialogHeader>
   505→            <DialogTitle className={`flex items-center gap-2 ${c.text}`}>
   506→              <div className={`${c.icon} p-1.5 rounded text-white`}>{icon}</div>
   507→              {editItem ? 'रेकॉर्ड संपादा' : 'नवीन रेकॉर्ड'}
   508→            </DialogTitle>
   509→          </DialogHeader>
   510→          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 py-4">
   511→            {formFields.map((f) => (
   512→              <div key={f.key} className={`space-y-1 ${f.colSpan === 2 ? 'sm:col-span-2' : ''}`}>
   513→                <Label className="text-sm">
   514→                  {f.label} {f.required && <span className="text-red-500">*</span>}
   515→                </Label>
   516→                {f.type === 'select' ? (
   517→                  <Select
   518→                    value={String(formData[f.key] || '')}
   519→                    onValueChange={(v) => setFormData((prev) => ({ ...prev, [f.key]: v }))}
   520→                  >
   521→                    <SelectTrigger>
   522→                      <SelectValue placeholder={f.placeholder || 'निवडा'} />
   523→                    </SelectTrigger>
   524→                    <SelectContent>
   525→                      {f.options?.map((opt) => (
   526→                        <SelectItem key={opt.value} value={opt.value}>
   527→                          {opt.label}
   528→                        </SelectItem>
   529→                      ))}
   530→                    </SelectContent>
   531→                  </Select>
   532→                ) : f.type === 'checkbox' ? (
   533→                  <div className="flex items-center gap-2 pt-2">
   534→                    <Checkbox
   535→                      checked={!!formData[f.key]}
   536→                      onCheckedChange={(checked) => setFormData((prev) => ({ ...prev, [f.key]: !!checked }))}
   537→                    />
   538→                    <span className="text-sm text-muted-foreground">{f.placeholder || 'होय'}</span>
   539→                  </div>
   540→                ) : f.type === 'textarea' ? (
   541→                  <Textarea
   542→                    value={String(formData[f.key] || '')}
   543→                    onChange={(e) => setFormData((prev) => ({ ...prev, [f.key]: e.target.value }))}

   546→                  />

   550→                    onChange={(e) => setFormData((prev) => ({ ...prev, [f.key]: e.target.value }))}

   564→              {saving ? <Loader2 className="h-4 w-4 mr-1 animate-spin" /> : <Save className="h-4 w-4 mr-1" />}
   565→              {saving ? 'जतन करत आहे...' : 'जतन करा'}
   566→            </Button>
   567→          </DialogFooter>
   568→        </DialogContent>
   569→      </Dialog>
   570→    </Card>
   571→  );
   572→}
