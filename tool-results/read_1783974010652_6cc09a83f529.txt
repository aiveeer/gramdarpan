   166→// ─── Namuna Key Parsing & Display ──────────────────────────────────────────
   167→
   168→/** Convert initialNamuna string like "5ka", "9ka", "20ka", "20kha", "26ka", "26kha" to the decimal num used in NAMUNA_DEFS */
   169→function parseNamunaKey(key: string): number {
   170→  const s = key.trim().toLowerCase();
   171→  // Sub-forms: 5ka → 5.1, 9ka → 9.1, 20ka → 20.1, 20kha → 20.2, 26ka → 26.1, 26kha → 26.2
   172→  if (s === '5ka') return 5.1;
   173→  if (s === '9ka') return 9.1;
   174→  if (s === '20ka') return 20.1;
   175→  if (s === '20kha') return 20.2;
   176→  if (s === '26ka') return 26.1;
   177→  if (s === '26kha') return 26.2;
   178→  return parseInt(s, 10);
   179→}
   180→
   181→/** Format namuna num for display: 5.1 → "५ क", 20.2 → "२० ख", etc. */
   182→function formatNamunaNum(num: number): string {
   183→  const marathiDigits = ['०','१','२','३','४','५','६','७','८','९'];
   184→  const toMr = (n: number) => String(n).split('').map(c => marathiDigits[parseInt(c)] ?? c).join('');
   185→  const base = Math.floor(num);
   186→  const sub = Math.round((num - base) * 10); // 1=क, 2=ख
   187→  const suffix = sub === 1 ? ' क' : sub === 2 ? ' ख' : '';
   188→  return toMr(base) + suffix;
   189→}
