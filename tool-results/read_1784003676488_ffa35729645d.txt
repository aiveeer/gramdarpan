   112→// Normalize a raw Prisma PropertyMaster object from /api/master to match PropertyInfo interface
   113→// The Prisma model uses `propertyNo` but the interface uses `propertyNumber`, etc.
   114→function normalizeProperty(raw: Record<string, unknown>): PropertyInfo {
   115→  // Parse boundaries JSON if it's a string
   116→  let boundaryData: Record<string, unknown> = {};
   117→  if (typeof raw.boundaries === 'string') {
   118→    try { boundaryData = JSON.parse(raw.boundaries); } catch { /* ignore */ }
   119→  }
   120→
   121→  // Parse floorInfo JSON if it's a string
   122→  let totalLength: number | undefined;
   123→  let totalWidth: number | undefined;
   124→  if (typeof raw.floorInfo === 'string') {
   125→    try {
   126→      const fi = JSON.parse(raw.floorInfo);
   127→      totalLength = fi.totalLength;
   128→      totalWidth = fi.totalWidth;
   129→    } catch { /* ignore */ }
   130→  }
   131→
   132→  const ward = raw.ward as Record<string, unknown> | null | undefined;
   133→  const road = raw.road as Record<string, unknown> | null | undefined;
   134→
   135→  return {
   136→    id: (raw.id as string) || '',
   137→    propertyNumber: (raw.propertyNo as string) || (raw.propertyNumber as string) || '',
   138→    area: raw.area != null ? Number(raw.area) : null,
   139→    builtUpArea: raw.builtUpArea != null ? Number(raw.builtUpArea) : null,
   140→    constructionType: (raw.constructionType as string) ?? null,
   141→    usageType: (raw.usageType as string) ?? (raw.propertyUse as string) ?? null,
   142→    citySurveyNo: (raw.citySurveyNo as string) ?? null,
   143→    boundaryEast: (boundaryData.boundaryEast as string) ?? undefined,
   144→    boundaryWest: (boundaryData.boundaryWest as string) ?? undefined,
   145→    boundarySouth: (boundaryData.boundarySouth as string) ?? undefined,
   146→    boundaryNorth: (boundaryData.boundaryNorth as string) ?? undefined,
   147→    lengthEast: boundaryData.lengthEast != null ? Number(boundaryData.lengthEast) : undefined,
   148→    widthEast: boundaryData.widthEast != null ? Number(boundaryData.widthEast) : undefined,
   149→    lengthWest: boundaryData.lengthWest != null ? Number(boundaryData.lengthWest) : undefined,
   150→    widthWest: boundaryData.widthWest != null ? Number(boundaryData.widthWest) : undefined,
   151→    lengthSouth: boundaryData.lengthSouth != null ? Number(boundaryData.lengthSouth) : undefined,
   152→    widthSouth: boundaryData.widthSouth != null ? Number(boundaryData.widthSouth) : undefined,
   153→    lengthNorth: boundaryData.lengthNorth != null ? Number(boundaryData.lengthNorth) : undefined,
   154→    widthNorth: boundaryData.widthNorth != null ? Number(boundaryData.widthNorth) : undefined,
   155→    totalLength,
   156→    totalWidth,
   157→    depreciationRate: raw.depreciationRate != null ? Number(raw.depreciationRate) : undefined,
   158→    usageFactor: raw.usageFactor != null ? Number(raw.usageFactor) : undefined,
   159→    taxRate: raw.taxRate != null ? Number(raw.taxRate) : undefined,
   160→    houseTax: raw.houseTax != null ? Number(raw.houseTax) : undefined,
   161→    lightTax: raw.lightTax != null ? Number(raw.lightTax) : undefined,
   162→    healthTax: raw.healthTax != null ? Number(raw.healthTax) : undefined,
   163→    waterTax: raw.waterTax != null ? Number(raw.waterTax) : undefined,
   164→    constructionDetails: (raw.constructionDetails as string) ?? undefined,
   165→    yearBuilt: (raw.yearBuilt as string) ?? undefined,
   166→    ward: ward ? {
   167→      wardNameMr: (ward.wardNameMr as string) ?? '',
   168→      wardNumber: (ward.wardNumber as string) ?? (ward.wardNo != null ? String(ward.wardNo) : ''),
   169→      wardName: (ward.wardName as string) ?? '',
   170→    } : undefined,
   171→    road: road ? {
   172→      roadNameMr: (road.roadNameMr as string) ?? '',
   173→      roadNumber: (road.roadNo as string) ?? (road.roadNumber as string) ?? '',
   174→      roadName: (road.roadName as string) ?? '',
   175→    } : undefined,
   176→    owners: Array.isArray(raw.owners)
   177→      ? raw.owners.map((o: Record<string, unknown>) => {
   178→          const ow = o.owner as Record<string, unknown> | null | undefined;
   179→          return {
   180→            owner: {
   181→              firstName: (ow?.firstName as string) ?? '',
   182→              lastName: (ow?.lastName as string) ?? '',
   183→              firstNameMr: (ow?.firstNameMr as string) ?? '',
   184→              lastNameMr: (ow?.lastNameMr as string) ?? '',
   185→            },
   186→            ownershipType: (o.ownershipType as string) ?? 'मालक',
   187→          };
   188→        })
   189→      : [],
   190→    taxRates: Array.isArray(raw.taxRates)
   191→      ? raw.taxRates.map((tr: Record<string, unknown>) => {
   192→          const tm = tr.taxMaster as Record<string, unknown> | null | undefined;
   193→          return {
   194→            taxMasterId: (tr.taxMasterId as string) ?? '',
   195→            rate: tr.rate != null ? Number(tr.rate) : 0,
   196→            taxMaster: {
   197→              name: (tm?.name as string) ?? '',
   198→              nameMarathi: (tm?.nameMarathi as string) ?? '',
   199→              isEnabled: tm?.isEnabled != null ? Boolean(tm.isEnabled) : true,
   200→              order: tm?.order != null ? Number(tm.order) : 0,
   201→              category: '',
   202→            },
   203→          };
   204→        })
   205→      : [],
   206→  };
   207→}

   209→// Normalize a raw Namuna8 record from /api/namuna8 to match Namuna8Record interface
   210→function normalizeRecord(raw: Record<string, unknown>): Namuna8Record {
   211→  return {
   212→    id: (raw.id as string) || '',
   213→    propertyId: (raw.propertyId as string) || '',
   214→    financialYear: (raw.financialYear as string) || '',
   215→    taxDetails: (raw.taxDetails as string) ?? '',
   216→    totalTax: raw.totalTax != null ? Number(raw.totalTax) : 0,
   217→    totalArea: raw.totalArea != null ? Number(raw.totalArea) : undefined,
   218→    landRate: raw.landRate != null ? Number(raw.landRate) : undefined,
   219→    buildingRate: raw.buildingRate != null ? Number(raw.buildingRate) : undefined,
   220→    constructionRate: raw.constructionRate != null ? Number(raw.constructionRate) : undefined,
   221→    depreciationRate: raw.depreciationRate != null ? Number(raw.depreciationRate) : undefined,
   222→    usageFactor: raw.usageFactor != null ? Number(raw.usageFactor) : undefined,
   223→    capitalValue: raw.capitalValue != null ? Number(raw.capitalValue) : undefined,
   224→    taxRatePercent: raw.taxRatePercent != null ? Number(raw.taxRatePercent) : undefined,
   225→    houseTaxAmt: raw.houseTaxAmt != null ? Number(raw.houseTaxAmt) : undefined,
   226→    lightTaxAmt: raw.lightTaxAmt != null ? Number(raw.lightTaxAmt) : undefined,
   227→    healthTaxAmt: raw.healthTaxAmt != null ? Number(raw.healthTaxAmt) : undefined,
   228→    waterTaxAmt: raw.waterTaxAmt != null ? Number(raw.waterTaxAmt) : undefined,
   229→    totalTaxAmt: raw.totalTaxAmt != null ? Number(raw.totalTaxAmt) : undefined,
   230→    constructionDetails: (raw.constructionDetails as string) ?? (raw.constructionDetailsStr as string) ?? undefined,
   231→    appealHouseTax: raw.appealHouseTax != null ? Number(raw.appealHouseTax) : undefined,
   232→    appealLightTax: raw.appealLightTax != null ? Number(raw.appealLightTax) : undefined,
   233→    appealHealthTax: raw.appealHealthTax != null ? Number(raw.appealHealthTax) : undefined,
   234→    appealWaterTax: raw.appealWaterTax != null ? Number(raw.appealWaterTax) : undefined,
   235→    appealTotalTax: raw.appealTotalTax != null ? Number(raw.appealTotalTax) : undefined,
   236→    remarks: (raw.remarks as string) ?? undefined,
   237→    property: normalizeProperty((raw.property as Record<string, unknown>) || {}),
   238→  };
   239→}

   283→  const [generatingAll, setGeneratingAll] = useState(false);
   284→  const [generateProgress, setGenerateProgress] = useState({ current: 0, total: 0 });
   285→
   286→  const fetchData = useCallback(async () => {
   287→    setLoading(true);
   288→    try {
   289→      const [recRes, propRes, villageRes] = await Promise.all([
   290→        fetch('/api/namuna8'),
   291→        fetch('/api/master?table=property'),
   292→        fetch('/api/master?table=village'),
   293→      ]);
   294→
   295→      // Safely parse and validate records - always ensure array
   296→      let normalizedRecords: Namuna8Record[] = [];
   297→      try {
   298→        const recData = await recRes.json();
   299→        if (Array.isArray(recData)) {
   300→          normalizedRecords = recData.map((r: Record<string, unknown>) => normalizeRecord(r));
   301→        }
   302→      } catch { /* ignore parse error, keep empty array */ }
   303→      setRecords(normalizedRecords);
   304→
   305→      // Safely parse and validate properties - always ensure array with normalization
   306→      let normalizedProps: PropertyInfo[] = [];
   307→      try {
   308→        const propData = await propRes.json();
   309→        if (Array.isArray(propData)) {
   310→          normalizedProps = propData.map((p: Record<string, unknown>) => normalizeProperty(p));
   311→        }
   312→      } catch { /* ignore parse error, keep empty array */ }
   313→      setProperties(normalizedProps);
   314→
   315→      // Safely parse village info
   316→      try {
   317→        const vd = await villageRes.json();
   318→        if (vd && !Array.isArray(vd)) setVillageInfo(vd as VillageInfo);
   319→      } catch { /* ignore parse error */ }
   320→    } catch {
   321→      toast({ title: 'त्रुटी', description: 'डेटा लोड अयशस्वी', variant: 'destructive' });
   322→    } finally {
   323→      setLoading(false);
   324→    }
